# Human

```tsx
<Human
  message="Review the implementation"
  onApprove={() => setPhase("continue")}
  onReject={() => setPhase("cancelled")}
>
  The agent has implemented the feature.
  Please review before continuing.
</Human>
```

## Props

| Prop | Type | Description |
|------|------|-------------|
| `id` | `string?` | Stable ID for resumability (optional - falls back to content hash) |
| `message` | `string?` | Short message shown to human |
| `children` | `ReactNode?` | Detailed content shown |
| `onApprove` | `() => void?` | Called on approval |
| `onReject` | `() => void?` | Called on rejection |

## Approval Workflow

```tsx
function ApprovalWorkflowBody() {
  const { db, reactiveDb } = useSmithers()
  const { data: phaseVal } = useQueryValue<string>(
    reactiveDb, "SELECT value FROM state WHERE key = 'phase'"
  )
  const phase = phaseVal ?? "implement"
  const setPhase = (p: string) => db.state.set('phase', p)

  return (
    <>
      <If condition={phase === "implement"}>
        <Claude onFinished={() => setPhase("review")}>
          Implement the feature.
        </Claude>
      </If>

      <If condition={phase === "review"}>
        <Human
          message="Review Implementation"
          onApprove={() => setPhase("deploy")}
          onReject={() => setPhase("implement")}
        >
          Review the changes in the diff.
        </Human>
      </If>

      <If condition={phase === "deploy"}>
        <Claude onFinished={() => setPhase("done")}>
          Deploy the changes.
        </Claude>
      </If>

      <If condition={phase === "cancelled"}>
        <Stop reason="Human rejected changes" />
      </If>
    </>
  )
}

function ApprovalWorkflow({ db, executionId }) {
  return (
    <SmithersProvider db={db} executionId={executionId}>
      <ApprovalWorkflowBody />
    </SmithersProvider>
  );
}
```
