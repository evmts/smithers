# Stop

```tsx
<Stop reason="Task completed successfully" />
```

## Props

| Prop | Type | Description |
|------|------|-------------|
| `reason` | `string?` | Reason for stopping |
| `children` | `ReactNode?` | Optional child content |

## Conditional Stop

```tsx
function ConditionalWorkflowBody() {
  const { db, reactiveDb } = useSmithers()
  const { data: statusVal } = useQueryValue<string>(
    reactiveDb, "SELECT value FROM state WHERE key = 'status'"
  )
  const status = statusVal ?? "running"
  const setStatus = (s: string) => db.state.set('status', s)

  return (
    <>
      <If condition={status === "running"}>
        <Claude
          onFinished={(result) => {
            if (result.output.includes("CRITICAL")) setStatus("error");
            else if (result.output.includes("DONE")) setStatus("complete");
          }}
        >
          Process the task.
        </Claude>
      </If>

      <If condition={status === "error"}>
        <Stop reason="Critical error detected" />
      </If>

      <If condition={status === "complete"}>
        <Stop reason="Task completed" />
      </If>
    </>
  )
}

function ConditionalWorkflow({ db, executionId }) {
  return (
    <SmithersProvider db={db} executionId={executionId}>
      <ConditionalWorkflowBody />
    </SmithersProvider>
  );
}
```
