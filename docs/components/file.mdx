# File Component

The `<File>` component writes or updates files during agent execution. It allows agents to produce file artifacts without requiring Claude tool calls, making file operations explicit and declarative in your agent workflows.

## Basic Usage

```tsx
import { File } from '@evmts/smithers'

<File path="output/report.md">
  # Analysis Report

  This is the generated report content.
</File>
```

## Props

<ParamField path="path" type="string" required>
  The file path to write to. Can be absolute or relative to the working directory.

  ```tsx
  <File path="./output/results.json">
    {JSON.stringify(data)}
  </File>
  ```
</ParamField>

<ParamField path="children" type="JSX.Element" required>
  The content to write to the file.
</ParamField>

<ParamField path="mode" type="'write' | 'append'" default="write">
  How to write the content. `write` overwrites the file, `append` adds to the end.

  ```tsx
  <File path="log.txt" mode="append">
    [{new Date().toISOString()}] Task completed
  </File>
  ```
</ParamField>

<ParamField path="encoding" type="string" default="utf-8">
  The file encoding to use.

  ```tsx
  <File path="output.txt" encoding="utf-8">
    Content here
  </File>
  ```
</ParamField>

<ParamField path="onWritten" type="(path: string) => void">
  Callback invoked after the file is successfully written.

  ```tsx
  <File
    path="report.md"
    onWritten={(path) => console.log(`Wrote ${path}`)}
  >
    Report content
  </File>
  ```
</ParamField>

<ParamField path="onError" type="(error: Error) => void">
  Callback invoked if writing fails.

  ```tsx
  <File
    path="/protected/file.txt"
    onError={(err) => console.error('Write failed:', err)}
  >
    Content
  </File>
  ```
</ParamField>

## Use Cases

### Generating Reports

Write analysis results to a file:

```tsx
import { createSignal } from 'solid-js'

function ReportAgent(props) {
  const [analysis, setAnalysis] = createSignal(null)

  return () => {
    const a = analysis()
    if (!a) {
      return (
        <Claude onFinished={setAnalysis} tools={[filesystem]}>
          Analyze {props.codebase} for code quality issues
        </Claude>
      )
    }

    return (
      <>
        <File path="reports/analysis.md">
          # Code Quality Report

          ## Summary
          {a.summary}

          ## Issues Found
          {a.issues.map(issue =>
            `- **${issue.severity}**: ${issue.message} (${issue.file}:${issue.line})`
          ).join('\n')}

          ## Recommendations
          {a.recommendations.join('\n')}
        </File>
        <Output>Report written to reports/analysis.md</Output>
        <Stop reason="Report generated" />
      </>
    )
  }
}
```

### Writing JSON Data

Save structured data:

```tsx
import { createSignal } from 'solid-js'

function DataExportAgent(props) {
  const [data, setData] = createSignal(null)

  return () => {
    const d = data()
    if (!d) {
      return (
        <Claude onFinished={setData}>
          Query the database: {props.query}
        </Claude>
      )
    }

    return (
      <File path="export/data.json">
        {JSON.stringify(d, null, 2)}
      </File>
    )
  }
}
```

### Appending to Logs

Add entries to a log file:

```tsx
import { createSignal } from 'solid-js'

function LoggingAgent(props) {
  const [result, setResult] = createSignal(null)

  return () => {
    const r = result()
    return (
      <>
        <File path="execution.log" mode="append">
          [{new Date().toISOString()}] Starting: {props.task}
        </File>

        <Claude onFinished={setResult}>
          Execute: {props.task}
        </Claude>

        {r && (
          <File path="execution.log" mode="append">
            [{new Date().toISOString()}] Completed: {props.task} - {r.status}
          </File>
        )}
      </>
    )
  }
}
```

### Multiple File Output

Generate multiple files from a single workflow:

```tsx
import { createSignal } from 'solid-js'

function DocumentationAgent(props) {
  const [docs, setDocs] = createSignal(null)

  return () => {
    const d = docs()
    if (!d) {
      return (
        <Claude onFinished={setDocs}>
          Generate documentation for: {props.modules.join(', ')}
        </Claude>
      )
    }

    return (
      <>
        {d.modules.map(mod => (
          <File key={mod.name} path={`docs/${mod.name}.md`}>
            # {mod.name}

            {mod.description}

            ## API
            {mod.api}
          </File>
        ))}

        <File path="docs/index.md">
          # Documentation Index

          {d.modules.map(mod =>
            `- [${mod.name}](./${mod.name}.md)`
          ).join('\n')}
        </File>

        <Output>Generated {d.modules.length + 1} documentation files</Output>
      </>
    )
  }
}
```

### Conditional File Writing

Write files based on conditions:

```tsx
function ConfigAgent(props) {
  return (
    <>
      <File path={`config/${props.environment}.json`}>
        {JSON.stringify(props.settings, null, 2)}
      </File>

      {props.environment === 'production' && (
        <File path="config/backup.json">
          {JSON.stringify({ ...props.settings, backup: true }, null, 2)}
        </File>
      )}
    </>
  )
}
```

## Combining with Other Components

File works well with Claude and control flow:

```tsx
import { createSignal } from 'solid-js'

function BuildAgent(props) {
  const [phase, setPhase] = createSignal('analyze')
  const [artifacts, setArtifacts] = createSignal({})

  return () => {
    const p = phase()
    const a = artifacts()

    if (p === 'analyze') {
      return (
        <Claude onFinished={(res) => { setArtifacts(res); setPhase('generate') }}>
          Analyze {props.project} and determine what files to generate
        </Claude>
      )
    }

    if (p === 'generate') {
      return (
        <>
          {a.files.map(file => (
            <File
              key={file.path}
              path={file.path}
              onWritten={() => console.log(`Generated: ${file.path}`)}
            >
              {file.content}
            </File>
          ))}

          <Human
            message="Review generated files before finalizing"
            onApprove={() => setPhase('complete')}
          >
            Generated {a.files.length} files
          </Human>
        </>
      )
    }

    return (
      <>
        <Output>Build complete! Generated {a.files.length} files.</Output>
        <Stop reason="Build complete" />
      </>
    )
  }
}
```

## Behavior

- File components are executed during the Ralph Wiggum loop
- Files are written after any preceding Claude executions complete
- If the parent directory doesn't exist, it will be created
- File writes are atomic - content is fully written or not at all
- In mock mode, file writes are simulated (no actual disk writes)

## Error Handling

Handle file write errors gracefully:

```tsx
import { createSignal } from 'solid-js'

function SafeFileWriter(props) {
  const [error, setError] = createSignal(null)

  return () => {
    const err = error()
    if (err) {
      return (
        <>
          <Output label="Error">Failed to write {props.path}: {err.message}</Output>
          <Stop reason="File write failed" />
        </>
      )
    }

    return (
      <File path={props.path} onError={setError}>
        {props.content}
      </File>
    )
  }
}
```
