# If

```tsx
<If condition={phase === "implement"}>
  <Phase name="Implementation">
    <Step name="implement">
      <Claude>Implement the feature.</Claude>
    </Step>
  </Phase>
</If>
```

## Props

| Prop | Type | Description |
|------|------|-------------|
| `condition` | `unknown` | Children render when truthy (any truthy/falsy value) |
| `children` | `ReactNode` | Content to render when condition is truthy |

## Multi-Phase Workflow

```tsx
function WorkflowBody() {
  const { db, reactiveDb } = useSmithers()
  const { data: phaseVal } = useQueryValue<string>(
    reactiveDb, "SELECT value FROM state WHERE key = 'phase'"
  )
  const phase = phaseVal ?? "research"
  const setPhase = (p: string) => db.state.set('phase', p)

  return (
    <>
      <If condition={phase === "research"}>
        <Phase name="Research">
          <Step name="research">
            <Claude onFinished={() => setPhase("implement")}>
              Research the codebase.
            </Claude>
          </Step>
        </Phase>
      </If>

      <If condition={phase === "implement"}>
        <Phase name="Implementation">
          <Step name="implement">
            <Claude onFinished={() => setPhase("test")}>
              Implement the feature.
            </Claude>
          </Step>
        </Phase>
      </If>

      <If condition={phase === "test"}>
        <Phase name="Testing">
          <Step name="test">
            <Claude onFinished={() => setPhase("done")}>
              Run tests.
            </Claude>
          </Step>
        </Phase>
      </If>
    </>
  )
}

function Workflow({ db, executionId }) {
  return (
    <SmithersProvider db={db} executionId={executionId} maxIterations={10}>
      <WorkflowBody />
    </SmithersProvider>
  )
}
```
